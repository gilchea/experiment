\chapter{Kiến thức chuẩn bị}

\section{Cơ sở giải tích lồi}

\begin{definition}[Tập lồi]
\label{def:convex_set}
Một tập hợp $C \subseteq \mathbb{R}^d$ được gọi là tập lồi (convex set) nếu với mọi $x, y \in C$ và với mọi $\theta \in [0, 1]$, điểm kết hợp lồi sau đây cũng thuộc về $C$:
\begin{equation}
\theta x + (1 - \theta)y \in C
\end{equation}
\end{definition}

Về mặt hình học, định nghĩa \ref{def:convex_set} có nghĩa là đoạn thẳng nối hai điểm bất kỳ nằm trong tập lồi phải hoàn toàn nằm trọn trong tập hợp đó. Lớp tập hợp này đảm bảo không gian tìm kiếm nghiệm không bị chia cắt hoặc chứa các khoảng lõm có thể cô lập nghiệm tối ưu toàn cục.

\begin{definition}[Hàm lồi]
\label{def:convex_function}
Một hàm số $f: C \to \mathbb{R}$ xác định trên tập lồi $C \subseteq \mathbb{R}^d$ được gọi là hàm lồi (convex function) nếu với mọi $x, y \in C$ và mọi $\theta \in [0, 1]$, ta luôn có bất đẳng thức:
\begin{equation}\label{eq:convex_func}
f(\theta x + (1 - \theta)y) \le \theta f(x) + (1 - \theta)f(y)
\end{equation}
Nếu bất đẳng thức diễn ra một cách nghiêm ngặt (dấu $<$ xảy ra với mọi $x \neq y$ và $\theta \in (0, 1)$), hàm $f$ được gọi là hàm lồi nghiêm ngặt (strictly convex).
\end{definition}

Ý nghĩa hình học của bất đẳng thức (\ref{eq:convex_func}) thể hiện rằng đoạn thẳng nối hai điểm bất kỳ trên đồ thị của hàm số luôn nằm phía trên hoặc trùng với đồ thị của chính hàm số đó. Tính lồi nghiêm ngặt đảm bảo tính duy nhất của nghiệm tối ưu nếu tồn tại.

Đối với các hàm số có tính chất toán học tốt hơn, cụ thể là lớp hàm khả vi liên tục, ta có thể đặc trưng tính lồi thông qua điều kiện bậc nhất như sau.

\begin{proposition}[Điều kiện bậc nhất của hàm lồi khả vi]
\label{prop:first_order_convex}
Cho $f$ là hàm khả vi liên tục trên miền xác định $\operatorname{dom} f$ là một tập lồi. Khi đó $f$ là hàm lồi khi và chỉ khi với mọi $x, y \in \operatorname{dom} f$, bất đẳng thức sau được thỏa mãn:
\begin{equation}\label{eq:first_order_convex}
f(y) \ge f(x) + \langle \nabla f(x), y - x \rangle
\end{equation}
\end{proposition}

Bất đẳng thức (\ref{eq:first_order_convex}) cung cấp một góc nhìn quan trọng: tiếp diện của hàm lồi tại một điểm bất kỳ luôn là một chặn dưới tuyến tính (global underestimator) của hàm số đó trên toàn bộ miền xác định. Đây chính là tính chất toán học cốt lõi cho phép các thuật toán trượt gradient dựa vào thông tin cục bộ tại một điểm để đưa ra hướng dịch chuyển tiệm cận về phía nghiệm tối ưu toàn cục.

\begin{definition}[Gradient và dưới gradient]
\label{def:gradient_subgradient}
\leavevmode
\begin{enumerate}
    \item Với hàm khả vi $f$, gradient của $f$ tại điểm $x$, ký hiệu $\nabla f(x)$, là vectơ chứa các đạo hàm riêng phần: $\nabla f(x) = \left( \frac{\partial f}{\partial x_1}, \dots, \frac{\partial f}{\partial x_d} \right)^T$.
    \item Một vectơ $g \in \mathbb{R}^d$ được gọi là một dưới gradient (subgradient) của hàm lồi $f$ tại điểm $x$ nếu với mọi $y \in \operatorname{dom} f$:
    \begin{equation}
    f(y) \ge f(x) + \langle g, y - x \rangle
    \end{equation}
    Tập hợp tất cả các dưới gradient của $f$ tại $x$ được ký hiệu là $\partial f(x)$ và gọi là dưới vi phân (subdifferential).
\end{enumerate}
\end{definition}

\begin{theorem}[Điều kiện tối ưu]
\label{thm:optimality_condition}
\leavevmode
\begin{enumerate}
    \item Nếu $f$ là hàm lồi khả vi trên $\mathbb{R}^d$, thì $x^*$ là nghiệm tối ưu toàn cục của bài toán $\min_{x \in \mathbb{R}^d} f(x)$ khi và chỉ khi:
    \begin{equation}\label{eq:opt_criterion_smooth}
    \nabla f(x^*) = 0
    \end{equation}
    \item Nếu $f$ là hàm lồi nhưng không khả vi, thì $x^*$ là nghiệm tối ưu toàn cục khi và chỉ khi:
    \begin{equation}\label{eq:opt_criterion_nonsmooth}
    0 \in \partial f(x^*)
    \end{equation}
\end{enumerate}
\end{theorem}

Sự khác biệt mang tính quyết định của tối ưu hóa lồi so với tối ưu hóa phi lồi nằm ở chỗ: điều kiện (\ref{eq:opt_criterion_smooth}) hoặc (\ref{eq:opt_criterion_nonsmooth}) không chỉ dừng lại ở việc xác định điểm cực trị cục bộ hay điểm yên ngựa, mà nó đảm bảo chắc chắn rằng $x^*$ là điểm tối tiểu toàn cục (global minimum). Do đó, mục tiêu của các thuật toán như GD, SGD hay SVRG thực chất là thiết lập một chuỗi các điểm xấp xỉ sao cho gradient (hoặc dưới gradient) tiến dần về mức không.

\section{Các khái niệm về độ trơn và tốc độ hội tụ}

\subsection{Hàm khả vi liên tục Lipschitz (L-smooth)}

Bên cạnh tính lồi, tốc độ hội tụ và độ ổn định của các giải thuật tối ưu hóa phụ thuộc rất lớn vào mức độ biến thiên của gradient của hàm mục tiêu. Khái niệm then chốt để kiểm soát sự biến thiên này là tính trơn Lipschitz của gradient.

\begin{definition}[Hàm trơn Lipschitz]
\label{def:l_smooth}
Một hàm số $f: \mathbb{R}^d \to \mathbb{R}$ khả vi liên tục được gọi là trơn Lipschitz với hằng số $L > 0$ (hay $L$-smooth) trên tập lồi $C$ nếu gradient của nó thỏa mãn điều kiện liên tục Lipschitz sau với mọi $x, y \in C$:
\begin{equation}\label{eq:lipschitz_gradient}
\|\nabla f(x) - \nabla f(y)\| \le L \|x - y\|
\end{equation}
\end{definition}

Hằng số $L$ (gọi là hằng số trơn Lipschitz) đóng vai trò định lượng giới hạn trên cho tốc độ thay đổi của vectơ gradient. Khi một hàm số thỏa mãn giả thiết (\ref{eq:lipschitz_gradient}), nó hàm ý rằng bề mặt của hàm mục tiêu không có các biến động quá đột ngột hay các vùng có độ cong tiến đến vô hạn \cite{schmidt2017minimizing}.

\begin{lemma}[Bổ đề Descent]
\label{lem:descent_lemma}
Nếu $f$ là hàm $L$-smooth trên $\mathbb{R}^d$, thì với mọi $x, y \in \mathbb{R}^d$, ta có bất đẳng thức sau:
\begin{equation}\label{eq:descent_lemma}
f(y) \le f(x) + \langle \nabla f(x), y - x \rangle + \frac{L}{2}\|y - x\|^2
\end{equation}
\end{lemma}

Bổ đề \ref{lem:descent_lemma} có ý nghĩa đặc biệt quan trọng trong việc thiết kế bước trượt của thuật toán. Nó đảm bảo rằng nếu thuật toán dịch chuyển từ vị trí $x$ theo hướng ngược chiều gradient với một kích thước bước học đủ nhỏ (cụ thể là hệ số bước học $\eta \le 1/L$), giá trị của hàm mục tiêu chắc chắn sẽ giảm xuống sau mỗi vòng lặp mà không xảy ra hiện tượng nhảy quá đà (overshooting) dẫn đến phân kỳ. Đây là điều kiện tiên quyết để chứng minh tính hội tụ của cả phương pháp Gradient Descent truyền thống lẫn các phương pháp ngẫu nhiên cải tiến sau này.

\subsection{Tốc độ hội tụ}

Để đánh giá và so sánh hiệu năng giữa các giải thuật tối ưu hóa, các nhà nghiên cứu sử dụng khái niệm tốc độ hội tụ (convergence rate), công cụ định lượng mối quan hệ giữa sai số tối ưu và số lượng vòng lặp thực thi của thuật toán.

\begin{definition}[Tốc độ hội tụ dưới tuyến tính]
\label{def:sublinear_convergence}
Một thuật toán được gọi là đạt tốc độ hội tụ dưới tuyến tính (sublinear convergence) nếu sai số hàm mục tiêu tại bước lặp $k$, ký hiệu $e_k = f(x_k) - f(x^*)$, giảm theo tỷ lệ nghịch với số bước lặp:
\begin{equation}
e_k \le \frac{C}{k^\alpha}
\end{equation}
trong đó $C > 0$ là hằng số và $\alpha > 0$ (thông thường $\alpha = 0.5$ hoặc $\alpha = 1$).
\end{definition}

Để đạt được độ chính xác sai số $\epsilon$, các thuật toán hội tụ dưới tuyến tính đòi hỏi số vòng lặp ở mức cấp độ rất lớn là $\mathcal{O}(1/\epsilon)$ hoặc $\mathcal{O}(1/\epsilon^2)$. Đây chính là tốc độ giới hạn mà các phương pháp trượt gradient ngẫu nhiên (SGD) tiêu chuẩn phải chấp nhận do tác động tiêu cực của nhiễu phương sai \cite{johnson2013accelerating}.

\begin{definition}[Tốc độ hội tụ tuyến tính]
\label{def:linear_convergence}
Một thuật toán được gọi là đạt tốc độ hội tụ tuyến tính (linear convergence) nếu tồn tại một hệ số co $\rho \in (0, 1)$ và hằng số $C > 0$ sao cho:
\begin{equation}
e_k \le C \cdot \rho^k
\end{equation}
\end{definition}

Mặc dù mang tên "tuyến tính", định nghĩa \ref{def:linear_convergence} trên thực tế đại diện cho tốc độ giảm sai số theo hàm mũ (exponential decay). Bất đẳng thức này chứng minh rằng sai số tối ưu được thu hẹp lại theo một tỷ lệ cố định sau mỗi bước lặp độc lập. Hệ quả là số lượng vòng lặp cần thiết để tiệm cận độ chính xác $\epsilon$ chỉ tăng trưởng ở mức logarithmic, tức là $\mathcal{O}(\log(1/\epsilon))$. Việc khôi phục thành công tốc độ hội tụ tuyến tính tiềm năng này cho môi trường tính toán ngẫu nhiên (với chi phí mỗi vòng lặp cố định $\mathcal{O}(1)$) chính là thành tựu kiệt xuất của hướng tiếp cận thu gọn phương sai dự báo mà thuật toán SVRG đại diện \cite{johnson2013accelerating}.

\section{Phương pháp Gradient Descent (GD)}

Phương pháp Gradient Descent (GD) truyền thống—hay còn gọi là phương pháp Gradient toàn cục (Full Gradient)—là giải thuật nền tảng nhất để giải bài toán tối ưu hóa không ràng buộc trong học máy. Nguyên lý hoạt động của GD dựa trên việc cập nhật tham số bằng cách dịch chuyển ngược hướng với vectơ gradient của toàn bộ hàm mục tiêu trên tập dữ liệu tại mỗi bước lặp.

\begin{definition}[Quy tắc cập nhật Gradient Descent]
\label{def:gd_update}
Cho hàm mục tiêu $P(w)$ khả vi. Quy tắc cập nhật của Gradient Descent tại bước lặp $t$ được xác định bởi:
\begin{equation}\label{eq:gd_update_rule}
w_{t+1} = w_t - \eta \nabla P(w_t)
\end{equation}
trong đó $\eta > 0$ là hệ số bước học (learning rate).
\end{definition}

\begin{theorem}[Hội tụ của GD cho hàm lồi trơn tổng quát]
\label{thm:gd_convex_smooth}
Giả sử $f$ là hàm lồi và $L$-smooth trên $\mathbb{R}^d$. Khi đó phương pháp Gradient Descent với kích thước bước học cố định $\eta = 1/L$ đạt được tốc độ hội tụ:
\begin{equation}
f(x_k) - f(x^*) \le \frac{2L \|x_0 - x^*\|^2}{k}
\end{equation}
nghĩa là thuật toán có tốc độ hội tụ dưới tuyến tính $\mathcal{O}(1/k)$.
\end{theorem}

\begin{theorem}[Hội tụ của GD cho hàm lồi mạnh]
\label{thm:gd_strongly_convex}
Giả sử $f$ là hàm $\gamma$-lồi mạnh và $L$-smooth trên $\mathbb{R}^d$. Khi đó phương pháp Gradient Descent với kích thước bước học cố định $\eta = 1/L$ đạt được tốc độ hội tụ tuyến tính:
\begin{equation}
f(x_k) - f(x^*) \le \left(1 - \frac{\gamma}{L}\right)^k \left[ f(x_0) - f(x^*) \right]
\end{equation}
\end{theorem}

Mối quan hệ giữa tốc độ hội tụ lý thuyết và chi phí thực thi thực tế của GD đã được Johnson và Zhang \cite{johnson2013accelerating} tóm tắt. Đối với hàm mục tiêu lồi mạnh và trơn Lipschitz, sai số tại bước lặp $t$ của GD tuân theo quy luật suy giảm hàm mũ dạng:
\begin{equation}
P(w_t) - P(w^*) \le \mathcal{O}\left(\left(1 - \frac{\gamma}{L}\right)^t\right)
\end{equation}
Trong đó $\gamma$ là hằng số lồi mạnh và $L$ là hằng số trơn Lipschitz. Tỷ số $\gamma/L$ liên quan mật thiết đến số điều kiện (condition number) của bài toán.

Mặc dù sở hữu tốc độ hội tụ tuyến tính lý tưởng về số vòng lặp, GD lại gặp phải một rào cản chí mạng về mặt chi phí tính toán khi đối mặt với dữ liệu quy mô lớn. Do hàm mục tiêu trong học máy được cấu thành từ tổng của $n$ hàm thành phần (tương ứng với $n$ mẫu dữ liệu), việc tính toán gradient toàn cục $\nabla P(w) = \frac{1}{n}\sum_{i=1}^n \nabla \psi_i(w)$ bắt buộc thuật toán phải duyệt qua toàn bộ tập huấn luyện. Hệ quả là chi phí máy tính cho mỗi bước cập nhật đơn lẻ đạt mức độ phức tạp $\mathcal{O}(n)$, khiến GD không khả thi cho các bài toán với $n$ lên đến hàng triệu hoặc hàng tỷ mẫu.

\section{Stochastic Gradient Descent (SGD) cổ điển}

Để phá vỡ thế bế tắc về mặt chi phí $\mathcal{O}(n)$ của GD, phương pháp Trượt Gradient Ngẫu nhiên (Stochastic Gradient Descent - SGD) đã được phát triển như một giải pháp thay thế hiệu quả cho các bài toán học máy quy mô lớn.

\begin{definition}[Quy tắc cập nhật Stochastic Gradient Descent]
\label{def:sgd_update}
Cho hàm mục tiêu $P(w) = \frac{1}{n}\sum_{i=1}^n \psi_i(w)$. Quy tắc cập nhật của SGD tại bước lặp $t$ được xác định bởi:
\begin{equation}\label{eq:sgd_update_rule}
w_{t+1} = w_t - \eta_t \nabla \psi_{i_t}(w_t)
\end{equation}
trong đó $\eta_t$ là hệ số bước học tại thời điểm $t$, và $i_t \in \{1, \dots, n\}$ là chỉ số mẫu được lấy ngẫu nhiên đều từ tập huấn luyện.
\end{definition}

\begin{proposition}[Tính không chệch của ước lượng SGD]
\label{prop:sgd_unbiased}
Với chỉ số mẫu $i_t$ được lấy ngẫu nhiên đều từ $\{1, \dots, n\}$, ước lượng gradient $\nabla \psi_{i_t}(w_t)$ là không chệch (unbiased), nghĩa là:
\begin{equation}
\mathbb{E}_{i_t} \left[ \nabla \psi_{i_t}(w_t) \right] = \frac{1}{n} \sum_{i=1}^n \nabla \psi_i(w_t) = \nabla P(w_t)
\end{equation}
\end{proposition}

Tính không chệch này đảm bảo rằng, mặc dù vectơ gradient ngẫu nhiên tại một bước lặp cụ thể có thể sai lệch đáng kể, nhưng tính trung bình trên phương diện thống kê, nó hoàn toàn trùng khớp với vectơ gradient toàn cục chính xác. Do đó, độ phức tạp tính toán cho mỗi bước lặp được giảm từ $\mathcal{O}(n)$ xuống mức cố định $\mathcal{O}(1)$, độc lập với kích thước tập dữ liệu.

\begin{assumption}[Điều kiện Robbins-Monro]
\label{ass:robbins_monro}
Để đảm bảo SGD hội tụ, chuỗi bước học $\eta_t$ phải thỏa mãn các điều kiện sau:
\begin{equation}
\sum_{t=1}^\infty \eta_t = \infty \quad \text{và} \quad \sum_{t=1}^\infty \eta_t^2 < \infty
\end{equation}
\end{assumption}

Điều kiện thứ nhất ($\sum \eta_t = \infty$) đảm bảo tổng các bước đi đủ lớn để thuật toán có thể vượt qua mọi khoảng cách và chạm đến nghiệm tối ưu từ bất kỳ điểm khởi đầu nào. Điều kiện thứ hai ($\sum \eta_t^2 < \infty$) đảm bảo tốc độ thu nhỏ của bước học đủ nhanh để triệt tiêu năng lượng nhiễu tích lũy từ các gradient ngẫu nhiên.

\begin{theorem}[Hội tụ dưới tuyến tính của SGD]
\label{thm:sgd_sublinear}
Dưới các giả thiết thích hợp (hàm lồi, gradient Lipschitz) và điều kiện Robbins-Monro \ref{ass:robbins_monro}, SGD đạt tốc độ hội tụ dưới tuyến tính:
\begin{equation}
\mathbb{E}[P(w_k) - P(w^*)] = \mathcal{O}(1/k)
\end{equation}
Ngay cả đối với hàm lồi mạnh, SGD cũng không thể đạt tốc độ hội tụ tuyến tính.
\end{theorem}

\subsection{Nguyên nhân căn bản của hội tụ chậm: phương sai không suy giảm}

Để tìm ra giải pháp cải tiến hiệu quả, việc làm rõ nguyên nhân gốc rễ dẫn đến tốc độ hội tụ chậm chạp của SGD là điều tối quan trọng.

\begin{definition}[Phương sai của ước lượng gradient]
\label{def:gradient_variance}
Phương sai của ước lượng gradient trong SGD được định nghĩa là:
\begin{equation}
\operatorname{Var}(\nabla \psi_{i_t}(w_t)) := \mathbb{E} \left[ \|\nabla \psi_{i_t}(w_t) - \nabla P(w_t)\|^2 \right]
\end{equation}
\end{definition}

\begin{proposition}[Phương sai không triệt tiêu]
\label{prop:variance_not_zero}
Trong phương pháp GD cổ điển, khi $w_t \to w^*$, ta có $\nabla P(w_t) \to 0$. Tuy nhiên, đối với SGD, ngay cả khi $w_t = w^*$, các gradient thành phần $\nabla \psi_i(w^*)$ nhìn chung vẫn khác không. Do đó:
\begin{equation}
\lim_{t \to \infty} \operatorname{Var}(\nabla \psi_{i_t}(w_t)) > 0
\end{equation}
nghĩa là phương sai của ước lượng gradient không triệt tiêu mà duy trì ở một mức hằng số dương.
\end{proposition}

Lượng phương sai không suy giảm này liên tục đẩy tham số ra xa khỏi nghiệm, buộc người dùng phải ép bước học $\eta_t$ tiến dần về 0 để duy trì sự ổn định. Khi bước học trở nên quá nhỏ, tốc độ dịch chuyển của thuật toán bị đình trệ, dẫn đến hiện tượng thắt nút cổ chai hội tụ dưới tuyến tính.

Trên các tập dữ liệu lớn ($n$ cực kỳ lớn) và trong các bài toán học máy quy mô công nghiệp nơi thời gian tính toán là tài nguyên bị giới hạn, SGD tỏ ra vượt trội hơn GD rất nhiều trong giai đoạn đầu để nhanh chóng tìm ra một nghiệm xấp xỉ hợp lý. Tuy nhiên, nếu ứng dụng đòi hỏi một nghiệm có độ chính xác cao, SGD sẽ bộc lộ nhược điểm và tốn rất nhiều thời gian ở giai đoạn sau. Sự thấu hiểu sâu sắc về mối quan hệ đánh đổi này chính là động lực trực tiếp dẫn dắt các nhà nghiên cứu đến với cuộc cách mạng tiếp theo: phát triển lớp thuật toán thu gọn phương sai (như SVRG) nhằm tích hợp hoàn hảo ưu điểm tốc độ của SGD và khả năng hội tụ nhanh của GD.
